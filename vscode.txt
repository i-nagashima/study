/* ショートカット
------------------------------------------------------------ */
# ズームイン
command + shift + -
# ズームアウト
command + -

# 単語移動
option + 左右

# 1行コピー (カットも同じ)
(何も選択しないで) command + c

# やり直し
command + z (戻る)
command + shift + z (進む)

# 複数範囲選択
command + d
さらに command + d で次の同じ単語
# 全部選択
command + F2

# 横断検索 (プロジェクト全体)
command + shift + f

# 行の入替
option + 上下

# ファイル移動(タブを移動)
command + option + 左右

# 置換
command + option + f

# サイドバーの表示／非表示
command + b

# ファイル検索
command + p  →  ファイル名を入力

# すべてのタブを保存
command + option + s

/* ショートカット(ソース系)
------------------------------------------------------------ */
# コメント
command + /
# ブロックコメント
option + shift + a

# 定義元の参照
F12
# ピーク(小窓)で定義元を参照
option + F12

# 使われているものを参照
shift + F12

# 間違っているコードのヒント
command + .

# シンボル名の変更(変数名の変更みたいな) ※ファイルをまたぐ
F2


/* コマンドパレット
------------------------------------------------------------ */
Mac : command + shift + p

/* 設定
------------------------------------------------------------ */
Mac : command + ,

/* 拡張機能
------------------------------------------------------------ */
Mac : command + shift + x

設定(command + ,)
