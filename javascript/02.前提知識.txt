const (ES6〜)
var (非推奨)

/* 関数
 * ------------------------------------------------------/
function fn(仮引数1, 仮引数2) {
    return 戻り値;
}

/* オブジェクト
 * ------------------------------------------------------/
名前(プロパティ, またはキー)と値(バリュー)をペアで管理する入れ物
let obj = {
    property1: 'Hello',
    property2: function() {

    },
    property3: {
        d: 'Bye'
    }
}

// プロパティへのアクセス方法
// ドット記法
obj.name
// ブラケット記法
obj['name']
