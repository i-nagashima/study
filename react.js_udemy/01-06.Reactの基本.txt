/* Create React App
------------------------------------------------------------ */
Reactのプロジェクトを作成する

# npxの場合
$ npx create-react-app <プロジェクト>

# npmの場合
$ npm init react-app <プロジェクト>

# yarnの場合
yarn create react-app <プロジェクト>

/* JSX記法のルールを知る
------------------------------------------------------------ */
return は１つのHTMLエレメントだけ
そのため<div></div>で囲うのたが、余計なdivがはいってしまうので
React.Fragment を使う
const App = () => {
  return (
    <React.Fragment>
      <h1>こんにちは</h1>
      <p>おげんきですか</p>
    </React.Fragment>
  );
}

さらに React.Fragment がめんどくさいので
<></>
で書いてもReact.Fragmentと同じになる
const App = () => {
  return (
    <>
      <h1>こんにちは</h1>
      <p>おげんきですか</p>
    </>
  );
}
